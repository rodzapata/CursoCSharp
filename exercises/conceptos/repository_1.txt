using Microsoft.Data.SqlClient;
using TuProyecto.Data;
using TuProyecto.Dtos;
using TuProyecto.Repositories.Interfaces;

namespace TuProyecto.Repositories
{
    public class EmpresaRepository : IEmpresaRepository
    {
        private readonly DbConnectionFactory _dbConnectionFactory;

        public EmpresaRepository(DbConnectionFactory dbConnectionFactory)
        {
            _dbConnectionFactory = dbConnectionFactory;
        }

        public async Task<EmpresaNitDto?> ObtenerEmpresaPorNitAsync(string nitCed, string tipo)
        {
            const string sql = @"
                SELECT 
                    a.tipodoc,
                    a.regimen,
                    a.nitced,
                    a.dv,
                    a.razon,
                    a.direccion,
                    a.telefono,
                    a.municipio,
                    b.dpto,
                    c.codDpto,
                    b.codmunicipio,
                    c.nombreDpto,
                    b.nombreCiudad,
                    d.num_aprobacion,
                    d.fecha1,
                    d.fecha2,
                    d.numfac1,
                    d.numfac2,
                    d.prefijo
                FROM grl_nita01 a
                LEFT JOIN municipios b ON a.municipio = b.codigo
                LEFT JOIN dpto c ON b.dpto = c.codigo
                LEFT JOIN empresa d ON a.nitced = d.nitced
                WHERE a.nitced = @NitCed
                  AND d.tipo = @Tipo;";

            using var connection = _dbConnectionFactory.CreateConnection();
            using var command = new SqlCommand(sql, connection);

            command.Parameters.AddWithValue("@NitCed", nitCed);
            command.Parameters.AddWithValue("@Tipo", tipo);

            await connection.OpenAsync();

            using var reader = await command.ExecuteReaderAsync();

            if (!await reader.ReadAsync())
                return null;

            return new EmpresaNitDto
            {
                TipoDoc = reader["tipodoc"] == DBNull.Value ? null : reader["tipodoc"].ToString(),
                Regimen = reader["regimen"] == DBNull.Value ? null : reader["regimen"].ToString(),
                NitCed = reader["nitced"] == DBNull.Value ? null : reader["nitced"].ToString(),
                Dv = reader["dv"] == DBNull.Value ? null : reader["dv"].ToString(),
                Razon = reader["razon"] == DBNull.Value ? null : reader["razon"].ToString(),
                Direccion = reader["direccion"] == DBNull.Value ? null : reader["direccion"].ToString(),
                Telefono = reader["telefono"] == DBNull.Value ? null : reader["telefono"].ToString(),
                Municipio = reader["municipio"] == DBNull.Value ? null : reader["municipio"].ToString(),
                Dpto = reader["dpto"] == DBNull.Value ? null : reader["dpto"].ToString(),
                CodDpto = reader["codDpto"] == DBNull.Value ? null : reader["codDpto"].ToString(),
                CodMunicipio = reader["codmunicipio"] == DBNull.Value ? null : reader["codmunicipio"].ToString(),
                NombreDpto = reader["nombreDpto"] == DBNull.Value ? null : reader["nombreDpto"].ToString(),
                NombreCiudad = reader["nombreCiudad"] == DBNull.Value ? null : reader["nombreCiudad"].ToString(),
                NumAprobacion = reader["num_aprobacion"] == DBNull.Value ? null : reader["num_aprobacion"].ToString(),
                Fecha1 = reader["fecha1"] == DBNull.Value ? null : Convert.ToDateTime(reader["fecha1"]),
                Fecha2 = reader["fecha2"] == DBNull.Value ? null : Convert.ToDateTime(reader["fecha2"]),
                NumFac1 = reader["numfac1"] == DBNull.Value ? null : reader["numfac1"].ToString(),
                NumFac2 = reader["numfac2"] == DBNull.Value ? null : reader["numfac2"].ToString(),
                Prefijo = reader["prefijo"] == DBNull.Value ? null : reader["prefijo"].ToString()
            };
        }
    }
}
