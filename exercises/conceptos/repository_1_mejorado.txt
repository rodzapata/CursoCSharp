using Microsoft.Data.SqlClient;
using System.Data;
using TuProyecto.Data;
using TuProyecto.Dtos;
using TuProyecto.Repositories.Interfaces;

namespace TuProyecto.Repositories
{
    public class EmpresaRepository : IEmpresaRepository
    {
        private readonly DbConnectionFactory _dbConnectionFactory;

        public EmpresaRepository(DbConnectionFactory dbConnectionFactory)
        {
            _dbConnectionFactory = dbConnectionFactory;
        }

        public async Task<EmpresaNitDto?> ObtenerEmpresaPorNitAsync(string nitCed, string tipo)
        {
            const string sql = @"
                SELECT 
                    a.tipodoc, a.regimen, a.nitced, a.dv, a.razon, a.direccion, 
                    a.telefono, a.municipio, b.dpto, c.codDpto, b.codmunicipio, 
                    c.nombreDpto, b.nombreCiudad, d.num_aprobacion, d.fecha1, 
                    d.fecha2, d.numfac1, d.numfac2, d.prefijo
                FROM grl_nita01 a
                LEFT JOIN municipios b ON a.municipio = b.codigo
                LEFT JOIN dpto c ON b.dpto = c.codigo
                LEFT JOIN empresa d ON a.nitced = d.nitced
                WHERE a.nitced = @NitCed
                  AND d.tipo = @Tipo;";

            using var connection = _dbConnectionFactory.CreateConnection();
            using var command = new SqlCommand(sql, connection);

            // Mejora 1: Tipado explícito de parámetros (evita problemas de rendimiento en SQL Server)
            command.Parameters.Add("@NitCed", SqlDbType.VarChar, 50).Value = nitCed;
            command.Parameters.Add("@Tipo", SqlDbType.VarChar, 20).Value = tipo;

            await connection.OpenAsync();

            using var reader = await command.ExecuteReaderAsync();

            if (!await reader.ReadAsync())
                return null;

            return new EmpresaNitDto
            {
                // Mejora 2: El uso de 'as string' extrae el texto o devuelve null automáticamente si es DBNull
                TipoDoc = reader["tipodoc"] as string,
                Regimen = reader["regimen"] as string,
                NitCed = reader["nitced"] as string,
                Dv = reader["dv"] as string,
                Razon = reader["razon"] as string,
                Direccion = reader["direccion"] as string,
                Telefono = reader["telefono"] as string,
                Municipio = reader["municipio"] as string,
                Dpto = reader["dpto"] as string,
                CodDpto = reader["codDpto"] as string,
                CodMunicipio = reader["codmunicipio"] as string,
                NombreDpto = reader["nombreDpto"] as string,
                NombreCiudad = reader["nombreCiudad"] as string,
                NumAprobacion = reader["num_aprobacion"] as string,

                // Mejora 3: Métodos de extensión limpios para tipos de valor nulos (DateTime)
                Fecha1 = reader.GetNullableDateTime("fecha1"),
                Fecha2 = reader.GetNullableDateTime("fecha2"),

                NumFac1 = reader["numfac1"] as string,
                NumFac2 = reader["numfac2"] as string,
                Prefijo = reader["prefijo"] as string
            };
        }
    }

    // Helper en el mismo archivo o en tu carpeta de Extensions
    public static class SqlDataReaderExtensions
    {
        public static DateTime? GetNullableDateTime(this SqlDataReader reader, string columnName)
        {
            var ordinal = reader.GetOrdinal(columnName);
            return reader.IsDBNull(ordinal) ? null : reader.GetDateTime(ordinal);
        }
    }
}
